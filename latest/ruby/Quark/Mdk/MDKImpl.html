    involved = ::DatawireQuarkCore::List.new([])
    idx = 0
    while ((idx) < ((suspects).size)) do
        node = (suspects)[idx]
        idx = (idx) + (1)
        (involved) << (node.toString())
        node.failure()
    end

    self.error("integration", text)
